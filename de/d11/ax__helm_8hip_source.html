160 template < const int LX >
161 int tune(void *w, void *u, void *dx, void *dy, void *dz,
162  void *dxt, void *dyt, void *dzt, void *h1,
163  void *g11, void *g22, void *g33, void *g12,
164  void *g13, void *g23, int *nelv, int *lx) {
165  hipEvent_t start,stop;
166  float time1,time2;
167  int retval;
168 
169  const dim3 nthrds_1d(1024, 1, 1);
170  const dim3 nblcks_1d((*nelv), 1, 1);
171  const dim3 nthrds_kstep((*lx), (*lx), 1);
172  const dim3 nblcks_kstep((*nelv), 1, 1);
173 
174  char *env_value = NULL;
175  char neko_log_buf[80];
176 
177  env_value=getenv("NEKO_AUTOTUNE");
178 
179  sprintf(neko_log_buf, "Autotune Ax helm (lx: %d)", *lx);
180  log_section(neko_log_buf);
181 
182  if(env_value) {
183  if( !strcmp(env_value,"1D") ) {
184  CASE_1D(LX);
185  sprintf(neko_log_buf,"Set by env : 1 (1D)");
186  log_message(neko_log_buf);
187  log_end_section();
188  return 1;
189  } else if( !strcmp(env_value,"KSTEP") ) {
190  CASE_KSTEP(LX);
191  sprintf(neko_log_buf,"Set by env : 2 (KSTEP)");
192  log_message(neko_log_buf);
193  log_end_section();
194  return 2;
195  } else {
196  sprintf(neko_log_buf, "Invalid value set for NEKO_AUTOTUNE");
197  log_error(neko_log_buf);
198  }
199  }
200 
201  hipEventCreate(&start);
202  hipEventCreate(&stop);
203 
204  hipEventRecord(start,0);
205 
206  for(int i = 0; i < 100; i++) {
207  CASE_1D(LX);
208  }
209 
210  hipEventRecord(stop,0);
211  hipEventSynchronize(stop);
212  hipEventElapsedTime(&time1, start, stop);
213 
214  hipEventRecord(start,0);
215 
216  for(int i = 0; i < 100; i++) {
217  CASE_KSTEP(LX);
218  }
219 
220  hipEventRecord(stop,0);
221  hipEventSynchronize(stop);
222  hipEventElapsedTime(&time2, start, stop);
223 
224  if(time1 < time2) {
225  retval = 1;
226  } else {
227  retval = 2;
228  }
229 
230  sprintf(neko_log_buf, "Chose : %d (%s)", retval,
231  (retval > 1 ? "KSTEP" : "1D"));
232  log_message(neko_log_buf);
233  log_end_section();
234  return retval;
235 }
