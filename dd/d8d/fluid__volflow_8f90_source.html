130  subroutine fluid_vol_flow_compute(this, u_res, v_res, w_res, p_res, &
131  ta1, ta2, ta3, ext_bdf, gs_Xh, c_Xh, rho, Re, bd, dt, &
132  bclst_dp, bclst_du, bclst_dv, bclst_dw, bclst_vel_res, &
133  Ax, ksp_prs, ksp_vel, pc_prs, pc_vel, niter)
134  class(fluid_volflow_t), intent(inout) :: this
135  type(field_t), intent(inout) :: u_res, v_res, w_res, p_res
136  type(field_t), intent(inout) :: ta1, ta2, ta3
137  type(coef_t), intent(inout) :: c_Xh
138  type(gs_t), intent(inout) :: gs_Xh
139  type(ext_bdf_scheme_t), intent(inout) :: ext_bdf
140  type(bc_list_t), intent(inout) :: bclst_dp, bclst_du, bclst_dv, bclst_dw
141  type(bc_list_t), intent(inout) :: bclst_vel_res
142  class(ax_t), intent(inout) :: Ax
143  class(ksp_t), intent(inout) :: ksp_prs, ksp_vel
144  class(pc_t), intent(inout) :: pc_prs, pc_vel
145  real(kind=rp), intent(inout) :: rho, re, bd, dt
146  integer, intent(in) :: niter
147  integer :: n, i
148  real(kind=rp) :: xlmin, xlmax
149  real(kind=rp) :: ylmin, ylmax
150  real(kind=rp) :: zlmin, zlmax
151  type(ksp_monitor_t) :: ksp_result
152 
153 
154  associate(msh => c_xh%msh, p_vol => this%p_vol, &
155  u_vol => this%u_vol, v_vol => this%v_vol, w_vol => this%w_vol)
156 
157  n = c_xh%dof%size()
158  xlmin = glmin(c_xh%dof%x, n)
159  xlmax = glmax(c_xh%dof%x, n)
160  ylmin = glmin(c_xh%dof%y, n) ! for Y!
161  ylmax = glmax(c_xh%dof%y, n)
162  zlmin = glmin(c_xh%dof%z, n) ! for Z!
163  zlmax = glmax(c_xh%dof%z, n)
164  if (this%flow_dir.eq.1) this%domain_length = xlmax - xlmin
165  if (this%flow_dir.eq.2) this%domain_length = ylmax - ylmin
166  if (this%flow_dir.eq.3) this%domain_length = zlmax - zlmin
167 
168  if (neko_bcknd_device .eq. 1) then
169  call device_cfill(c_xh%h1_d, 1.0_rp/rho, n)
170  call device_rzero(c_xh%h2_d, n)
171  else
172  do i = 1, n
173  c_xh%h1(i,1,1,1) = 1.0_rp / rho
174  c_xh%h2(i,1,1,1) = 0.0_rp
175  end do
176  end if
177  c_xh%ifh2 = .false.
178 
179  ! Compute pressure
180 
181  if (this%flow_dir .eq. 1) then
182  call cdtp(p_res%x, c_xh%h1, c_xh%drdx, c_xh%dsdx, c_xh%dtdx, c_xh)
183  end if
184 
185  if (this%flow_dir .eq. 2) then
186  call cdtp(p_res%x, c_xh%h1, c_xh%drdy, c_xh%dsdy, c_xh%dtdy, c_xh)
187  end if
188 
189  if (this%flow_dir .eq. 3) then
190  call cdtp(p_res%x, c_xh%h1, c_xh%drdz, c_xh%dsdz, c_xh%dtdz, c_xh)
191  end if
192 
193  call gs_op(gs_xh, p_res, gs_op_add)
194  call bc_list_apply_scalar(bclst_dp, p_res%x, n)
195  call pc_prs%update()
196  ksp_result = ksp_prs%solve(ax, p_vol, p_res%x, n, &
197  c_xh, bclst_dp, gs_xh, niter)
198 
199  ! Compute velocity
200 
201  call opgrad(u_res%x, v_res%x, w_res%x, p_vol%x, c_xh)
202 
203  if ((neko_bcknd_hip .eq. 1) .or. (neko_bcknd_cuda .eq. 1) .or. &
204  (neko_bcknd_opencl .eq. 1)) then
205  call device_opchsign(u_res%x_d, v_res%x_d, w_res%x_d, msh%gdim, n)
206  call device_copy(ta1%x_d, c_xh%B_d, n)
207  call device_copy(ta2%x_d, c_xh%B_d, n)
208  call device_copy(ta3%x_d, c_xh%B_d, n)
209  else
210  call opchsign(u_res%x, v_res%x, w_res%x, msh%gdim, n)
211  call copy(ta1%x, c_xh%B, n)
212  call copy(ta2%x, c_xh%B, n)
213  call copy(ta3%x, c_xh%B, n)
214  end if
215  call bc_list_apply_vector(bclst_vel_res,&
216  ta1%x, ta2%x, ta3%x, n)
217 
218  ! add forcing
219 
220  if (neko_bcknd_device .eq. 1) then
221  if (this%flow_dir .eq. 1) then
222  call device_add2(u_res%x_d, ta1%x_d, n)
223  else if (this%flow_dir .eq. 2) then
224  call device_add2(v_res%x_d, ta2%x_d, n)
225  else if (this%flow_dir .eq. 3) then
226  call device_add2(w_res%x_d, ta3%x_d, n)
227  end if
228  else
229  if (this%flow_dir .eq. 1) then
230  call add2(u_res%x, ta1%x, n)
231  else if (this%flow_dir .eq. 2) then
232  call add2(v_res%x, ta2%x, n)
233  else if (this%flow_dir .eq. 3) then
234  call add2(w_res%x, ta3%x, n)
235  end if
236  end if
237 
238  if (neko_bcknd_device .eq. 1) then
239  call device_cfill(c_xh%h1_d, (1.0_rp / re), n)
240  call device_cfill(c_xh%h2_d, rho * (bd / dt), n)
241  else
242  do i = 1, n
243  c_xh%h1(i,1,1,1) = (1.0_rp / re)
244  c_xh%h2(i,1,1,1) = rho * (bd / dt)
245  end do
246  end if
247  c_xh%ifh2 = .true.
248 
249  call gs_op(gs_xh, u_res, gs_op_add)
250  call gs_op(gs_xh, v_res, gs_op_add)
251  call gs_op(gs_xh, w_res, gs_op_add)
252 
253  call bc_list_apply_vector(bclst_vel_res,&
254  u_res%x, v_res%x, w_res%x, n)
255  call pc_vel%update()
256 
257  ksp_result = ksp_vel%solve(ax, u_vol, u_res%x, n, &
258  c_xh, bclst_du, gs_xh, niter)
259  ksp_result = ksp_vel%solve(ax, v_vol, v_res%x, n, &
260  c_xh, bclst_dv, gs_xh, niter)
261  ksp_result = ksp_vel%solve(ax, w_vol, w_res%x, n, &
262  c_xh, bclst_dw, gs_xh, niter)
263 
264  if (neko_bcknd_device .eq. 1) then
265  if (this%flow_dir .eq. 1) then
266  this%base_flow = &
267  device_glsc2(u_vol%x_d, c_xh%B_d, n) / this%domain_length
268  end if
269 
270  if (this%flow_dir .eq. 2) then
271  this%base_flow = &
272  device_glsc2(v_vol%x_d, c_xh%B_d, n) / this%domain_length
273  end if
274 
275  if (this%flow_dir .eq. 3) then
276  this%base_flow = &
277  device_glsc2(w_vol%x_d, c_xh%B_d, n) / this%domain_length
278  end if
279  else
280  if (this%flow_dir .eq. 1) then
281  this%base_flow = glsc2(u_vol%x, c_xh%B, n) / this%domain_length
282  end if
283 
284  if (this%flow_dir .eq. 2) then
285  this%base_flow = glsc2(v_vol%x, c_xh%B, n) / this%domain_length
286  end if
287 
288  if (this%flow_dir .eq. 3) then
289  this%base_flow = glsc2(w_vol%x, c_xh%B, n) / this%domain_length
290  end if
291  end if
292  end associate
293 
294  end subroutine fluid_vol_flow_compute

305  subroutine fluid_vol_flow(this, u, v, w, p, u_res, v_res, w_res, p_res, &
306  ta1, ta2, ta3, c_Xh, gs_Xh, ext_bdf, rho, Re, dt, &
307  bclst_dp, bclst_du, bclst_dv, bclst_dw, bclst_vel_res, &
308  Ax, ksp_prs, ksp_vel, pc_prs, pc_vel, niter)
309 
310  class(fluid_volflow_t), intent(inout) :: this
311  type(field_t), intent(inout) :: u, v, w, p
312  type(field_t), intent(inout) :: u_res, v_res, w_res, p_res
313  type(field_t), intent(inout) :: ta1, ta2, ta3
314  type(coef_t), intent(inout) :: c_Xh
315  type(gs_t), intent(inout) :: gs_Xh
316  type(ext_bdf_scheme_t), intent(inout) :: ext_bdf
317  real(kind=rp), intent(inout) :: rho, re, dt
318  type(bc_list_t), intent(inout) :: bclst_dp, bclst_du, bclst_dv, bclst_dw
319  type(bc_list_t), intent(inout) :: bclst_vel_res
320  class(ax_t), intent(inout) :: Ax
321  class(ksp_t), intent(inout) :: ksp_prs, ksp_vel
322  class(pc_t), intent(inout) :: pc_prs, pc_vel
323  integer, intent(in) :: niter
324  real(kind=rp) :: ifcomp, flow_rate, xsec
325  real(kind=rp) :: current_flow, delta_flow, base_flow, scale
326  integer :: n, ierr
327 
328  associate(u_vol => this%u_vol, v_vol => this%v_vol, &
329  w_vol => this%w_vol, p_vol => this%p_vol)
330 
331  n = c_xh%dof%size()
332 
333  ! If either dt or the backwards difference coefficient change,
334  ! then recompute base flow solution corresponding to unit forcing:
335 
336  ifcomp = 0.0_rp
337 
338  if (dt .ne. this%dtlag .or. ext_bdf%bdf(1) .ne. this%bdlag) then
339  ifcomp = 1.0_rp
340  end if
341 
342  this%dtlag = dt
343  this%bdlag = ext_bdf%bdf(1)
344 
345  call mpi_allreduce(mpi_in_place, ifcomp, 1, &
346  mpi_real_precision, mpi_sum, neko_comm, ierr)
347 
348  if (ifcomp .gt. 0d0) then
349  call this%compute(u_res, v_res, w_res, p_res, &
350  ta1, ta2, ta3, ext_bdf, gs_xh, c_xh, rho, re, ext_bdf%bdf(1), dt, &
351  bclst_dp, bclst_du, bclst_dv, bclst_dw, bclst_vel_res, &
352  ax, ksp_vel, ksp_prs, pc_prs, pc_vel, niter)
353  end if
354 
355  if (neko_bcknd_device .eq. 1) then
356  if (this%flow_dir .eq. 1) then
357  current_flow = &
358  device_glsc2(u%x_d, c_xh%B_d, n) / this%domain_length ! for X
359  else if (this%flow_dir .eq. 2) then
360  current_flow = &
361  device_glsc2(v%x_d, c_xh%B_d, n) / this%domain_length ! for Y
362  else if (this%flow_dir .eq. 3) then
363  current_flow = &
364  device_glsc2(w%x_d, c_xh%B_d, n) / this%domain_length ! for Z
365  end if
366  else
367  if (this%flow_dir .eq. 1) then
368  current_flow = glsc2(u%x, c_xh%B, n) / this%domain_length ! for X
369  else if (this%flow_dir .eq. 2) then
370  current_flow = glsc2(v%x, c_xh%B, n) / this%domain_length ! for Y
371  else if (this%flow_dir .eq. 3) then
372  current_flow = glsc2(w%x, c_xh%B, n) / this%domain_length ! for Z
373  end if
374  end if
375 
376  if (this%avflow) then
377  xsec = c_xh%volume / this%domain_length
378  flow_rate = this%flow_rate*xsec
379  endif
380 
381  delta_flow = flow_rate - current_flow
382  scale = delta_flow / this%base_flow
383 
384  if (neko_bcknd_device .eq. 1) then
385  call device_add2s2(u%x_d, u_vol%x_d, scale, n)
386  call device_add2s2(v%x_d, v_vol%x_d, scale, n)
387  call device_add2s2(w%x_d, w_vol%x_d, scale, n)
388  call device_add2s2(p%x_d, p_vol%x_d, scale, n)
389  else
390  call add2s2(u%x, u_vol%x, scale, n)
391  call add2s2(v%x, v_vol%x, scale, n)
392  call add2s2(w%x, w_vol%x, scale, n)
393  call add2s2(p%x, p_vol%x, scale, n)
394  end if
395  end associate
396 
397  end subroutine fluid_vol_flow
