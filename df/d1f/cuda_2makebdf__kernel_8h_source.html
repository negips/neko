36 __global__ void makebdf_kernel(const T * __restrict__ ulag1,
37  const T * __restrict__ ulag2,
38  const T * __restrict__ vlag1,
39  const T * __restrict__ vlag2,
40  const T * __restrict__ wlag1,
41  const T * __restrict__ wlag2,

44  T * __restrict__ bfz,
45  const T * __restrict__ u,
46  const T * __restrict__ v,
47  const T * __restrict__ w,
48  const T * __restrict__ B,
49  const T rho,
50  const T dt,
51  const T bd2,
52  const T bd3,
53  const T bd4,
54  const int nbd,
55  const int n) {
56 
57  const int idx = blockIdx.x * blockDim.x + threadIdx.x;
58  const int str = blockDim.x * gridDim.x;
59 
60  for (int i = idx; i < n; i += str) {
61  T tb1_val = u[i] * B[i] * bd2;
62  T tb2_val = v[i] * B[i] * bd2;
63  T tb3_val = w[i] * B[i] * bd2;
64 
65  T ta1_val = ulag1[i] * B[i] * bd3;
66  T ta2_val = vlag1[i] * B[i] * bd3;
67  T ta3_val = wlag1[i] * B[i] * bd3;
68 
69  tb1_val += ta1_val;
70  tb2_val += ta2_val;
71  tb3_val += ta3_val;
72 
73  if (nbd == 3) {
74  tb1_val += ulag2[i] * B[i] * bd4;
75  tb2_val += vlag2[i] * B[i] * bd4;
76  tb3_val += wlag2[i] * B[i] * bd4;
77  }
78 
79  bfx[i] = bfx[i] + tb1_val * (rho / dt);
80  bfy[i] = bfy[i] + tb2_val * (rho / dt);
81  bfz[i] = bfz[i] + tb3_val * (rho / dt);
82  }
83 
84 }
