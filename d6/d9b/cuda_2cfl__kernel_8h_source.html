39 template< typename T>
40 __inline__ __device__ T cfl_reduce_warp(T val) {
41  val = fmax(val, __shfl_down_sync(0xffffffff, val, 16));
42  val = fmax(val, __shfl_down_sync(0xffffffff, val, 8));
43  val = fmax(val, __shfl_down_sync(0xffffffff, val, 4));
44  val = fmax(val, __shfl_down_sync(0xffffffff, val, 2));
45  val = fmax(val, __shfl_down_sync(0xffffffff, val, 1));
46  return val;
47 }

57  const int str = blockDim.x * gridDim.x;

84 template< typename T, const int LX, const int CHUNKS >
85 __global__ void cfl_kernel(const T dt,
86  const T * __restrict__ u,
87  const T * __restrict__ v,
88  const T * __restrict__ w,
89  const T * __restrict__ drdx,
90  const T * __restrict__ dsdx,
91  const T * __restrict__ dtdx,
92  const T * __restrict__ drdy,
93  const T * __restrict__ dsdy,
94  const T * __restrict__ dtdy,
95  const T * __restrict__ drdz,
96  const T * __restrict__ dsdz,
97  const T * __restrict__ dtdz,
98  const T * __restrict__ dr_inv,
99  const T * __restrict__ ds_inv,
100  const T * __restrict__ dt_inv,
101  const T * __restrict__ jacinv,
102  T * __restrict__ cfl_h) {
103 
104  int i,j,k;
105 
106  const int e = blockIdx.x;
107  const int iii = threadIdx.x;
108  const int nchunks = (LX * LX * LX - 1) / CHUNKS + 1;
109  const unsigned int lane = threadIdx.x % warpSize;
110  const unsigned int wid = threadIdx.x / warpSize;
111 
112  __shared__ T shu[LX * LX * LX];
113  __shared__ T shv[LX * LX * LX];
114  __shared__ T shw[LX * LX * LX];
115 
116  __shared__ T shdr_inv[LX];
117  __shared__ T shds_inv[LX];
118  __shared__ T shdt_inv[LX];
119 
120  __shared__ T shjacinv[LX * LX * LX];
121 
122  __shared__ T shared[32];
123 
124  if (iii < LX) {
125  shdr_inv[iii] = dr_inv[iii];
126  shds_inv[iii] = ds_inv[iii];
127  shdt_inv[iii] = dt_inv[iii];
128  }
129 
130  j = iii;
131  while(j < (LX * LX * LX)) {
132  shu[j] = u[j + e * LX * LX * LX];
133  shv[j] = v[j + e * LX * LX * LX];
134  shw[j] = w[j + e * LX * LX * LX];
135 
136  shjacinv[j] = jacinv[j + e * LX * LX * LX];
137 
138  j = j + CHUNKS;
139  }
140 
141  __syncthreads();
142 
143  T cfl_tmp = 0.0;
144  for (int n = 0; n < nchunks; n++) {
145  const int ijk = iii + n * CHUNKS;
146  const int jk = ijk / LX;
147  i = ijk - jk * LX;
148  k = jk / LX;
149  j = jk - k * LX;
150  if ( i < LX && j < LX && k < LX) {
151  const T cflr = fabs( dt * ( ( shu[ijk] * drdx[ijk + e * LX * LX * LX]
152  + shv[ijk] * drdy[ijk + e * LX * LX * LX]
153  * shw[ijk] * drdz[ijk + e * LX * LX * LX]
154  ) * shjacinv[ijk]) * shdr_inv[i]);
155  const T cfls = fabs( dt * ( ( shu[ijk] * dsdx[ijk + e * LX * LX * LX]
156  + shv[ijk] * dsdy[ijk + e * LX * LX * LX]
157  + shw[ijk] * dsdz[ijk + e * LX * LX * LX]
158  ) * shjacinv[ijk]) * shds_inv[j]);
159  const T cflt = fabs( dt * ( ( shu[ijk] * dtdx[ijk + e * LX * LX * LX]
160  + shv[ijk] * dtdy[ijk + e * LX * LX * LX]
161  + shw[ijk] * dtdz[ijk + e * LX * LX * LX]
162  ) * shjacinv[ijk]) * shdt_inv[k]);
163 
164  cfl_tmp = fmax(cflr + cfls + cflt, cfl_tmp);
165 
166  }
167  }
168 
169  cfl_tmp = cfl_reduce_warp<T>(cfl_tmp);
170  if (lane == 0)
171  shared[wid] = cfl_tmp;
172  __syncthreads();
173 
174  cfl_tmp = (threadIdx.x < blockDim.x / warpSize) ? shared[lane] : 0;
175  if (wid == 0)
176  cfl_tmp = cfl_reduce_warp<T>(cfl_tmp);
177 
178  if (threadIdx.x == 0)
179  cfl_h[blockIdx.x] = cfl_tmp;
180 }
