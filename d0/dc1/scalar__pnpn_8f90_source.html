108  call this%free()

112

225  subroutine scalar_pnpn_step(this, t, tstep, ext_bdf)
226  class(scalar_pnpn_t), intent(inout) :: this
227  real(kind=rp), intent(inout) :: t
228  type(ext_bdf_scheme_t), intent(inout) :: ext_bdf
229  integer, intent(inout) :: tstep
230  integer :: n, niter
231  type(ksp_monitor_t) :: ksp_results(1)
232  n = this%dm_Xh%size()
233  niter = 1000
234 
235  associate(u => this%u, v => this%v, w => this%w, s => this%s, &
236  ds => this%ds, &
237  ta1 => this%ta1, &
238  wa1 => this%wa1, &
239  s_res =>this%s_res, &
240  ax => this%Ax, f_xh => this%f_Xh, xh => this%Xh, &
241  c_xh => this%c_Xh, dm_xh => this%dm_Xh, gs_xh => this%gs_Xh, &
242  slag => this%slag, &
243  params => this%params, msh => this%msh, res => this%res, &
244  sumab => this%sumab, &
245  makeext => this%makeext, makebdf => this%makebdf)
246 
247  ! evaluate the source term and scale with the mass matrix
248  call f_xh%eval()
249 
250  if (neko_bcknd_device .eq. 1) then
251  call device_col2(f_xh%s_d, c_xh%B_d, n)
252  else
253  call col2(f_xh%s, c_xh%B, n)
254  end if
255 
256  call this%adv%apply_scalar(this%u, this%v, this%w, this%s, f_xh%s, &
257  xh, this%c_Xh, dm_xh%size())
258 
259  call makeext%compute_scalar(ta1, this%abx1, this%abx2, f_xh%s, &
260  params%rho, ext_bdf%ext, n)
261 
262  call makebdf%compute_scalar(ta1, this%wa1, slag, f_xh%s, s, c_xh%B, &
263  params%rho, params%dt, ext_bdf%bdf, ext_bdf%nbd, n)
264 
265  call slag%update()
269  call this%bc_apply()
270 
271  ! compute scalar residual
272  call res%compute(ax, s, s_res, f_xh, c_xh, msh, xh, params%Pr, &
273  params%Re, params%rho, ext_bdf%bdf(1), params%dt, &
274  dm_xh%size())
275 
276  call gs_op(gs_xh, s_res, gs_op_add)
277 
278  call bc_list_apply_scalar(this%bclst_ds,&
279  s_res%x, dm_xh%size())
280 
281  if (tstep .gt. 5 .and. params%proj_vel_dim .gt. 0) then
282  call this%proj_s%project_on(s_res%x, c_xh, n)
283  end if
284 
285  call this%pc%update()
286  ksp_results(1) = this%ksp%solve(ax, ds, s_res%x, n, &
287  c_xh, this%bclst_ds, gs_xh, niter)
288 
289  if (tstep .gt. 5 .and. params%proj_vel_dim .gt. 0) then
290  call this%proj_s%project_back(ds%x, ax, c_xh, &
291  this%bclst_ds, gs_xh, n)
292  end if
293 
294  if (neko_bcknd_device .eq. 1) then
295  call device_add2s2(s%x_d, ds%x_d, 1.0_rp, n)
296  else
297  call add2s2(s%x, ds%x, 1.0_rp, n)
298  end if
299 
300  call scalar_step_info(tstep, t, params%dt, ksp_results)
301 
302  end associate
303  end subroutine scalar_pnpn_step
