40 __global__ void ax_helm_kernel(T * __restrict__ w,
41  const T * __restrict__ u,
42  const T * __restrict__ dx,
43  const T * __restrict__ dy,
44  const T * __restrict__ dz,
45  const T * __restrict__ h1,

47  const T * __restrict__ g22,

51  const T * __restrict__ g23) {
52 
53  __shared__ T shdx[LX * LX];
54  __shared__ T shdy[LX * LX];
55  __shared__ T shdz[LX * LX];
56 
57  __shared__ T shu[LX * LX];
58  __shared__ T shur[LX * LX];
59  __shared__ T shus[LX * LX];
60 
61  T ru[LX];
62  T rw[LX];
63  T rut;
64 
65  const int e = blockIdx.x;
66  const int j = threadIdx.y;
67  const int i = threadIdx.x;
68  const int ij = i + j*LX;
69  const int ele = e*LX*LX*LX;
70 
71  shdx[ij] = dx[ij];
72  shdy[ij] = dy[ij];
73  shdz[ij] = dz[ij];
74 
75 #pragma unroll
76  for(int k = 0; k < LX; ++k){
77  ru[k] = u[ij + k*LX*LX + ele];
78  rw[k] = 0.0;
79  }
80 
81 
82  __syncthreads();
83 #pragma unroll
84  for (int k = 0; k < LX; ++k){
85  const int ijk = ij + k*LX*LX;
86  const T G00 = g11[ijk+ele];
87  const T G11 = g22[ijk+ele];
88  const T G22 = g33[ijk+ele];
89  const T G01 = g12[ijk+ele];
90  const T G02 = g13[ijk+ele];
91  const T G12 = g23[ijk+ele];
92  const T H1 = h1[ijk+ele];
93  T ttmp = 0.0;
94  shu[ij] = ru[k];
95  for (int l = 0; l < LX; l++){
96  ttmp += shdz[k+l*LX] * ru[l];
97  }
98  __syncthreads();
99 
100  T rtmp = 0.0;
101  T stmp = 0.0;
102 #pragma unroll
103  for (int l = 0; l < LX; l++){
104  rtmp += shdx[i+l*LX] * shu[l+j*LX];
105  stmp += shdy[j+l*LX] * shu[i+l*LX];
106  }
107  shur[ij] = H1
108  * (G00 * rtmp
109  + G01 * stmp
110  + G02 * ttmp);
111  shus[ij] = H1
112  * (G01 * rtmp
113  + G11 * stmp
114  + G12 * ttmp);
115  rut = H1
116  * (G02 * rtmp
117  + G12 * stmp
118  + G22 * ttmp);
119 
120  __syncthreads();
121 
122  T wijke = 0.0;
123 #pragma unroll
124  for (int l = 0; l < LX; l++){
125  wijke += shur[l+j*LX] * shdx[l+i*LX];
126  rw[l] += rut * shdz[k+l*LX];
127  wijke += shus[i+l*LX] * shdy[l + j*LX];
128  }
129  rw[k] += wijke;
130  }
131 #pragma unroll
132  for (int k = 0; k < LX; ++k){
133  w[ij + k*LX*LX + ele] = rw[k];
134  }
135 }
