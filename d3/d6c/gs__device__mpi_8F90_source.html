187  subroutine gs_device_mpi_buf_init(this, pe_order, dof_stack, mark_dupes)
188  class(gs_device_mpi_buf_t), intent(inout) :: this
189  integer, allocatable, intent(inout) :: pe_order(:)
190  type(stack_i4_t), allocatable, intent(inout) :: dof_stack(:)
191  logical, intent(in) :: mark_dupes
192  integer, allocatable :: dofs(:)
193  integer :: i, j, total
194  integer(c_size_t) :: sz
195  type(htable_i4_t) :: doftable
196  integer :: dupe, marked, k
197  real(c_rp) :: rp_dummy
198  integer(c_int32_t) :: i4_dummy
199 
200  call device_mpi_init_reqs(size(pe_order), this%reqs)
201 
202  allocate(this%ndofs(size(pe_order)))
203  allocate(this%offset(size(pe_order)))
204 
205  total = 0
206  do i = 1, size(pe_order)
207  this%ndofs(i) = dof_stack(pe_order(i))%size()
208  this%offset(i) = total
209  total = total + this%ndofs(i)
210  end do
211 
212  this%total = total
213 
214  sz = c_sizeof(rp_dummy) * total
215  call device_alloc(this%buf_d, sz)
216 
217  sz = c_sizeof(i4_dummy) * total
218  call device_alloc(this%dof_d, sz)
219 
220  if (mark_dupes) call doftable%init(2*total)
221  allocate(dofs(total))
222 
223  ! Copy from dof_stack into dofs, optionally marking duplicates with doftable
224  marked = 0
225  do i = 1, size(pe_order)
226  ! %array() breaks on cray
227  select type (arr => dof_stack(pe_order(i))%data)
228  type is (integer)
229  do j = 1, this%ndofs(i)
230  k = this%offset(i) + j
231  if (mark_dupes) then
232  if (doftable%get(arr(j), dupe) .eq. 0) then
233  if (dofs(dupe) .gt. 0) then
234  dofs(dupe) = -dofs(dupe)
235  marked = marked + 1
236  end if
237  dofs(k) = -arr(j)
238  marked = marked + 1
239  else
240  call doftable%set(arr(j), k)
241  dofs(k) = arr(j)
242  end if
243  else
244  dofs(k) = arr(j)
245  end if
246  end do
247  end select
248  end do
249 
250  call device_memcpy(dofs, this%dof_d, total, host_to_device)
251 
252  deallocate(dofs)
253  call doftable%free()
254 
255  end subroutine gs_device_mpi_buf_init

400  subroutine gs_device_mpi_nbwait(this, u, n, op)
401  class(gs_device_mpi_t), intent(inout) :: this
402  integer, intent(in) :: n

455  end do

458 
459  ! Sync non-blocking streams
