132  subroutine device_alloc(x_d, s)
133  type(c_ptr), intent(inout) :: x_d

135  integer :: ierr
136 #ifdef HAVE_HIP
137  if (hipmalloc(x_d, s) .ne. hipsuccess) then
138  call neko_error('Memory allocation on device failed')
139  end if

142  call neko_error('Memory allocation on device failed')

144 #elif HAVE_OPENCL
145  x_d = clcreatebuffer(glb_ctx, cl_mem_read_write, s, c_null_ptr, ierr)
146  if (ierr .ne. cl_success) then
147  call neko_error('Memory allocation on device failed')
148  end if

347  subroutine device_memcpy_common(ptr_h, x_d, s, dir, sync_device)
348  type(c_ptr), intent(inout) :: ptr_h
349  type(c_ptr), intent(inout) :: x_d
350  integer(c_size_t), intent(in) :: s
351  integer, intent(in), value :: dir
352  logical, intent(in) :: sync_device
353 #ifdef HAVE_HIP
354  if (sync_device) then
355  if (dir .eq. host_to_device) then
356  if (hipmemcpy(x_d, ptr_h, s, hipmemcpyhosttodevice) &
357  .ne. hipsuccess) then
358  call neko_error('Device memcpy (host-to-device) failed')
359  end if
360  else if (dir .eq. device_to_host) then
361  if (hipmemcpy(ptr_h, x_d, s, hipmemcpydevicetohost) &
362  .ne. hipsuccess) then
363  call neko_error('Device memcpy (device-to-host) failed')
364  end if
365  else if (dir .eq. device_to_device) then
366  if (hipmemcpy(ptr_h, x_d, s, hipmemcpydevicetodevice) &
367  .ne. hipsuccess) then
368  call neko_error('Device memcpy (device-to-device) failed')
369  end if
370  else
371  call neko_error('Device memcpy failed (invalid direction')
372  end if
373  else
374  if (dir .eq. host_to_device) then
375  if (hipmemcpyasync(x_d, ptr_h, s, hipmemcpyhosttodevice) &
376  .ne. hipsuccess) then
377  call neko_error('Device memcpy async (host-to-device) failed')
378  end if
379  else if (dir .eq. device_to_host) then
380  if (hipmemcpyasync(ptr_h, x_d, s, hipmemcpydevicetohost) &
381  .ne. hipsuccess) then
382  call neko_error('Device memcpy async (device-to-host) failed')
383  end if
384  else if (dir .eq. device_to_device) then
385  if (hipmemcpyasync(ptr_h, x_d, s, hipmemcpydevicetodevice) &
386  .ne. hipsuccess) then
387  call neko_error('Device memcpy async (device-to-device) failed')
388  end if
389  else
390  call neko_error('Device memcpy failed (invalid direction')
391  end if
392  end if
393 #elif HAVE_CUDA
394  if (sync_device) then
395  if (dir .eq. host_to_device) then
396  if (cudamemcpy(x_d, ptr_h, s, cudamemcpyhosttodevice) &
397  .ne. cudasuccess) then
398  call neko_error('Device memcpy (host-to-device) failed')
399  end if
400  else if (dir .eq. device_to_host) then
401  if (cudamemcpy(ptr_h, x_d, s, cudamemcpydevicetohost) &
402  .ne. cudasuccess) then
403  call neko_error('Device memcpy (device-to-host) failed')
404  end if
405  else if (dir .eq. device_to_device) then
406  if (cudamemcpy(ptr_h, x_d, s, cudamemcpydevicetodevice) &
407  .ne. cudasuccess) then
408  call neko_error('Device memcpy (device-to-device) failed')
409  end if
410  else
411  call neko_error('Device memcpy failed (invalid direction')
412  end if
413  else
414  if (dir .eq. host_to_device) then
415  if (cudamemcpyasync(x_d, ptr_h, s, cudamemcpyhosttodevice) &
416  .ne. cudasuccess) then
417  call neko_error('Device memcpy async (host-to-device) failed')
418  end if
419  else if (dir .eq. device_to_host) then
420  if (cudamemcpyasync(ptr_h, x_d, s, cudamemcpydevicetohost) &
421  .ne. cudasuccess) then
422  call neko_error('Device memcpy async (device-to-host) failed')
423  end if
424  else if (dir .eq. device_to_device) then
425  if (cudamemcpyasync(ptr_h, x_d, s, cudamemcpydevicetodevice) &
426  .ne. cudasuccess) then
427  call neko_error('Device memcpy async (device-to-device) failed')
428  end if
429  else
430  call neko_error('Device memcpy failed (invalid direction')
431  end if
432  end if
433 #elif HAVE_OPENCL
434  if (sync_device) then
435  if (dir .eq. host_to_device) then
436  if (clenqueuewritebuffer(glb_cmd_queue, x_d, cl_true, 0_i8, s, ptr_h, &
437  0, c_null_ptr, c_null_ptr) .ne. cl_success) then
438  call neko_error('Device memcpy (host-to-device) failed')
439  end if
440  else if (dir .eq. device_to_host) then
441  if (clenqueuereadbuffer(glb_cmd_queue, x_d, cl_true, 0_i8, s, ptr_h, &
442  0, c_null_ptr, c_null_ptr) .ne. cl_success) then
443  call neko_error('Device memcpy (host-to-device) failed')
444  end if
445  else if (dir .eq. device_to_device) then
446  if (clenqueuecopybuffer(glb_cmd_queue, x_d, ptr_h, 0_i8, 0_i8, s, &
447  0, c_null_ptr, c_null_ptr) .ne. cl_success) then
448  call neko_error('Device memcpy (device-to-device) failed')
449  end if
450  else
451  call neko_error('Device memcpy failed (invalid direction')
452  end if
453  else
454  if (dir .eq. host_to_device) then
455  if (clenqueuewritebuffer(glb_cmd_queue, x_d, cl_false, 0_i8, s, ptr_h, &
456  0, c_null_ptr, c_null_ptr) .ne. cl_success) then
457  call neko_error('Device memcpy (host-to-device) failed')
458  end if
459  else if (dir .eq. device_to_host) then
460  if (clenqueuereadbuffer(glb_cmd_queue, x_d, cl_false, 0_i8, s, ptr_h, &
461  0, c_null_ptr, c_null_ptr) .ne. cl_success) then
462  call neko_error('Device memcpy (host-to-device) failed')
463  end if
464  else if (dir .eq. device_to_device) then
465  if (clenqueuecopybuffer(glb_cmd_queue, x_d, ptr_h, 0_i8, 0_i8, s, &
466  0, c_null_ptr, c_null_ptr) .ne. cl_success) then
467  call neko_error('Device memcpy (device-to-device) failed')
468  end if
469  else
470  call neko_error('Device memcpy failed (invalid direction')
471  end if
472  end if
473 #endif
474  end subroutine device_memcpy_common
