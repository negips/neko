173  subroutine cuda_device_name(name)
174  character(len=*), intent(inout) :: name
175  character(kind=c_char, len=8192), target :: prop
176  integer :: end_pos
177 
178  !
179  ! Yes this is an ugly hack!
180  ! Since we're only interested in the device name (first 256 bytes)
181  ! we pass down a large enough chunk of memory to the cuda runtime
182  ! and extract what we need later on
183  !
184  ! This will of course break if sizeof(cudaDeviceProp) > 8192
185  !
186 
187  if (cudagetdeviceproperties(c_loc(prop), 0) .ne. cudasuccess) then
188  call neko_error('Failed to query device')
189  end if
190 
191  end_pos = scan(prop(1:256), c_null_char)
192  if(end_pos .ge. 2) then
193  name(1:end_pos-1) = prop(1:end_pos-1)
194  endif
195  end subroutine cuda_device_name
