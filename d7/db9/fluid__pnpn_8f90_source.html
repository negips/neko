119  subroutine fluid_pnpn_init(this, msh, lx, param)
120  class(fluid_pnpn_t), target, intent(inout) :: this
121  type(mesh_t), target, intent(inout) :: msh
122  integer, intent(inout) :: lx
123  type(param_t), target, intent(inout) :: param
124  character(len=15), parameter :: scheme = 'Modular (Pn/Pn)'
125 
126  call this%free()
127 
128  ! Setup velocity and pressure fields on the space \f$ Xh \f$
129  call this%scheme_init(msh, lx, param, .true., .true., scheme)
130 
131  ! Setup backend dependent Ax routines
132  call ax_helm_factory(this%ax)
133 
134  ! Setup backend dependent prs residual routines
135  call pnpn_prs_res_factory(this%prs_res)
136 
137  ! Setup backend dependent vel residual routines
138  call pnpn_vel_res_factory(this%vel_res)
139 
140  ! Setup backend dependent summation of AB/BDF
141  call rhs_maker_sumab_fctry(this%sumab)
142 
143  ! Setup backend dependent summation of extrapolation scheme
144  call rhs_maker_ext_fctry(this%makeabf)
145 
146  ! Setup backend depenent contributions to F from lagged BD terms
147  call rhs_maker_bdf_fctry(this%makebdf)
148 
149  ! Initialize variables specific to this plan
150  associate(xh_lx => this%Xh%lx, xh_ly => this%Xh%ly, xh_lz => this%Xh%lz, &
151  dm_xh => this%dm_Xh, nelv => this%msh%nelv)
152 
153  call field_init(this%p_res, dm_xh, "p_res")
154  call field_init(this%u_res, dm_xh, "u_res")
155  call field_init(this%v_res, dm_xh, "v_res")
156  call field_init(this%w_res, dm_xh, "w_res")
157  call field_init(this%abx1, dm_xh, "abx1")
158  call field_init(this%aby1, dm_xh, "aby1")
159  call field_init(this%abz1, dm_xh, "abz1")
160 
161  call field_init(this%abx2, dm_xh, "abx2")
162  call field_init(this%aby2, dm_xh, "aby2")
163  call field_init(this%abz2, dm_xh, "abz2")
164 
165  call field_init(this%u_e, dm_xh, 'u_e')
166  call field_init(this%v_e, dm_xh, 'v_e')
167  call field_init(this%w_e, dm_xh, 'w_e')
168 
169  call field_init(this%wa1, dm_xh, 'wa1')
170  call field_init(this%wa2, dm_xh, 'wa2')
171  call field_init(this%wa3, dm_xh, 'wa3')
172 
173  call field_init(this%ta1, dm_xh, 'ta1')
174  call field_init(this%ta2, dm_xh, 'ta2')
175  call field_init(this%ta3, dm_xh, 'ta3')
176 
177  call field_init(this%du, dm_xh, 'du')
178  call field_init(this%dv, dm_xh, 'dv')
179  call field_init(this%dw, dm_xh, 'dw')
180  call field_init(this%dp, dm_xh, 'dp')
181 
182  call field_init(this%work1, dm_xh, 'work1')
183  call field_init(this%work2, dm_xh, 'work2')
184 
185  call this%ulag%init(this%u, 2)
186  call this%vlag%init(this%v, 2)
187  call this%wlag%init(this%w, 2)
188 
189  end associate
190 
191  ! Initialize velocity surface terms in pressure rhs
192  call this%bc_prs_surface%init(this%dm_Xh)
193  call this%bc_prs_surface%mark_zone(msh%inlet)
194  call this%bc_prs_surface%mark_zones_from_list(msh%labeled_zones,&
195  'v', this%params%bc_labels)
196  call this%bc_prs_surface%finalize()
197  call this%bc_prs_surface%set_coef(this%c_Xh)
198  ! Initialize symmetry surface terms in pressure rhs
199  call this%bc_sym_surface%init(this%dm_Xh)
200  call this%bc_sym_surface%mark_zone(msh%sympln)
201  call this%bc_sym_surface%mark_zones_from_list(msh%labeled_zones,&
202  'sym', this%params%bc_labels)
203  call this%bc_sym_surface%finalize()
204  call this%bc_sym_surface%set_coef(this%c_Xh)
205  ! Initialize dirichlet bcs for velocity residual
206  call this%bc_vel_res_non_normal%init(this%dm_Xh)
207  call this%bc_vel_res_non_normal%mark_zone(msh%outlet_normal)
208  call this%bc_vel_res_non_normal%mark_zones_from_list(msh%labeled_zones,&
209  'on', this%params%bc_labels)
210  call this%bc_vel_res_non_normal%mark_zones_from_list(msh%labeled_zones,&
211  'on+dong', this%params%bc_labels)
212  call this%bc_vel_res_non_normal%finalize()
213  call this%bc_vel_res_non_normal%init_msk(this%c_Xh)
214 
215  call this%bc_dp%init(this%dm_Xh)
216  call this%bc_dp%mark_zones_from_list(msh%labeled_zones,&
217  'on+dong', this%params%bc_labels)
218  call this%bc_dp%mark_zones_from_list(msh%labeled_zones,&
219  'o+dong', this%params%bc_labels)
220  call this%bc_dp%finalize()
221  call this%bc_dp%set_g(0.0_rp)
222  call bc_list_init(this%bclst_dp)
223  call bc_list_add(this%bclst_dp, this%bc_dp)
224  !Add 0 prs bcs
225  call bc_list_add(this%bclst_dp, this%bc_prs)
226 
227  call this%bc_vel_res%init(this%dm_Xh)
228  call this%bc_vel_res%mark_zone(msh%inlet)
229  call this%bc_vel_res%mark_zone(msh%wall)
230  call this%bc_vel_res%mark_zones_from_list(msh%labeled_zones,&
231  'v', this%params%bc_labels)
232  call this%bc_vel_res%mark_zones_from_list(msh%labeled_zones,&
233  'w', this%params%bc_labels)
234  call this%bc_vel_res%finalize()
235  call this%bc_vel_res%set_g(0.0_rp)
236  call bc_list_init(this%bclst_vel_res)
237  call bc_list_add(this%bclst_vel_res, this%bc_vel_res)
238  call bc_list_add(this%bclst_vel_res, this%bc_vel_res_non_normal)
239  call bc_list_add(this%bclst_vel_res, this%bc_sym)
240 
241  !Initialize bcs for u, v, w velocity components
242  call bc_list_init(this%bclst_du)
243  call bc_list_add(this%bclst_du,this%bc_sym%bc_x)
244  call bc_list_add(this%bclst_du,this%bc_vel_res_non_normal%bc_x)
245  call bc_list_add(this%bclst_du, this%bc_vel_res)
246 
247  call bc_list_init(this%bclst_dv)
248  call bc_list_add(this%bclst_dv,this%bc_sym%bc_y)
249  call bc_list_add(this%bclst_dv,this%bc_vel_res_non_normal%bc_y)
250  call bc_list_add(this%bclst_dv, this%bc_vel_res)
251 
252  call bc_list_init(this%bclst_dw)
253  call bc_list_add(this%bclst_dw,this%bc_sym%bc_z)
254  call bc_list_add(this%bclst_dw,this%bc_vel_res_non_normal%bc_z)
255  call bc_list_add(this%bclst_dw, this%bc_vel_res)
256 
257  !Intialize projection space thingy
258  if (param%proj_prs_dim .gt. 0) then
259  call this%proj_prs%init(this%dm_Xh%size(), param%proj_prs_dim)
260  end if
261 
262  if (param%proj_vel_dim .gt. 0) then
263  call this%proj_u%init(this%dm_Xh%size(), param%proj_vel_dim)
264  call this%proj_v%init(this%dm_Xh%size(), param%proj_vel_dim)
265  call this%proj_w%init(this%dm_Xh%size(), param%proj_vel_dim)
266  end if
267 
268  ! Add lagged term to checkpoint
269  call this%chkp%add_lag(this%ulag, this%vlag, this%wlag)
270  call advection_factory(this%adv, this%c_Xh, param%dealias, param%lxd)
271 
272  call this%vol_flow%init(this%dm_Xh, param)
273 
274  end subroutine fluid_pnpn_init

356  subroutine fluid_pnpn_step(this, t, tstep, ext_bdf)
357  class(fluid_pnpn_t), intent(inout) :: this
358  real(kind=rp), intent(inout) :: t
359  type(ext_bdf_scheme_t), intent(inout) :: ext_bdf
360  integer, intent(inout) :: tstep
361  integer :: n, niter
362  type(ksp_monitor_t) :: ksp_results(4)
363  n = this%dm_Xh%size()
364  niter = 1000
365 
366  associate(u => this%u, v => this%v, w => this%w, p => this%p, &
367  du => this%du, dv => this%dv, dw => this%dw, dp => this%dp, &
368  u_e => this%u_e, v_e => this%v_e, w_e => this%w_e, &
369  ta1 => this%ta1, ta2 => this%ta2, ta3 => this%ta3, &
370  wa1 => this%wa1, wa2 => this%wa2, wa3 => this%wa3, &
371  u_res =>this%u_res, v_res => this%v_res, w_res => this%w_res, &
372  p_res => this%p_res, ax => this%Ax, f_xh => this%f_Xh, xh => this%Xh, &
373  c_xh => this%c_Xh, dm_xh => this%dm_Xh, gs_xh => this%gs_Xh, &
374  ulag => this%ulag, vlag => this%vlag, wlag => this%wlag, &
375  params => this%params, msh => this%msh, prs_res => this%prs_res, &
376  vel_res => this%vel_res, sumab => this%sumab, &
377  makeabf => this%makeabf, makebdf => this%makebdf)
378 
379 
380  call sumab%compute_fluid(u_e, v_e, w_e, u, v, w, &
381  ulag, vlag, wlag, ext_bdf%ext, ext_bdf%nab)
382 
383  call f_xh%eval()
384 
385  if (neko_bcknd_device .eq. 1) then
386  call device_opcolv(f_xh%u_d, f_xh%v_d, f_xh%w_d, c_xh%B_d, msh%gdim, n)
387  else
388  call opcolv(f_xh%u, f_xh%v, f_xh%w, c_xh%B, msh%gdim, n)
389  end if
390 
391  call this%adv%apply(this%u, this%v, this%w, &
392  f_xh%u, f_xh%v, f_xh%w, &
393  xh, this%c_Xh, dm_xh%size())
394 
395  call makeabf%compute_fluid(ta1, ta2, ta3,&
396  this%abx1, this%aby1, this%abz1,&
397  this%abx2, this%aby2, this%abz2, &
398  f_xh%u, f_xh%v, f_xh%w,&
399  params%rho, ext_bdf%ext, n)
400 
401  call makebdf%compute_fluid(ta1, ta2, ta3, this%wa1, this%wa2, this%wa3,&
402  ulag, vlag, wlag, f_xh%u, f_xh%v, f_xh%w, &
403  u, v, w, c_xh%B, params%rho, params%dt, &
404  ext_bdf%bdf, ext_bdf%nbd, n)
405 
406  call ulag%update()
407  call vlag%update()
408  call wlag%update()
412  call this%bc_apply_vel()
413  call this%bc_apply_prs()
414 
415  ! compute pressure
416  call prs_res%compute(p, p_res, u, v, w, u_e, v_e, w_e, &
417  ta1, ta2, ta3, wa1, wa2, wa3, &
418  this%work1, this%work2, f_xh, &
419  c_xh, gs_xh, this%bc_prs_surface, &
420  this%bc_sym_surface, ax, ext_bdf%bdf(1), &
421  params%dt, params%Re, params%rho)
422 
423  call gs_op(gs_xh, p_res, gs_op_add)
424  call bc_list_apply_scalar(this%bclst_dp, p_res%x, p%dof%size())
425 
426  if( tstep .gt. 5 .and. params%proj_prs_dim .gt. 0) then
427  call this%proj_prs%project_on(p_res%x, c_xh, n)
428  call this%proj_prs%log_info('Pressure')
429  end if
430 
431  call this%pc_prs%update()
432  call profiler_start_region('Pressure')
433  ksp_results(1) = this%ksp_prs%solve(ax, dp, p_res%x, n, c_xh, &
434  this%bclst_dp, gs_xh, niter)
435  call profiler_end_region
436 
437  if( tstep .gt. 5 .and. params%proj_prs_dim .gt. 0) then
438  call this%proj_prs%project_back(dp%x, ax, c_xh, &
439  this%bclst_dp, gs_xh, n)
440  end if
441 
442  if (neko_bcknd_device .eq. 1) then
443  call device_add2(p%x_d, dp%x_d,n)
444  else
445  call add2(p%x, dp%x,n)
446  end if
447 
448 
449  ! compute velocity
450  call vel_res%compute(ax, u, v, w, &
451  u_res, v_res, w_res, &
452  p, ta1, ta2, ta3, &
453  f_xh, c_xh, msh, xh, &
454  params%Re, params%rho, ext_bdf%bdf(1), &
455  params%dt, dm_xh%size())
456 
457  call gs_op(gs_xh, u_res, gs_op_add)
458  call gs_op(gs_xh, v_res, gs_op_add)
459  call gs_op(gs_xh, w_res, gs_op_add)
460 
461  call bc_list_apply_vector(this%bclst_vel_res,&
462  u_res%x, v_res%x, w_res%x, dm_xh%size())
463 
464  if (tstep .gt. 5 .and. params%proj_vel_dim .gt. 0) then
465  call this%proj_u%project_on(u_res%x, c_xh, n)
466  call this%proj_v%project_on(v_res%x, c_xh, n)
467  call this%proj_w%project_on(w_res%x, c_xh, n)
468  end if
469 
470  call this%pc_vel%update()
471 
472  call profiler_start_region("Velocity")
473  ksp_results(2) = this%ksp_vel%solve(ax, du, u_res%x, n, &
474  c_xh, this%bclst_du, gs_xh, niter)
475  ksp_results(3) = this%ksp_vel%solve(ax, dv, v_res%x, n, &
476  c_xh, this%bclst_dv, gs_xh, niter)
477  ksp_results(4) = this%ksp_vel%solve(ax, dw, w_res%x, n, &
478  c_xh, this%bclst_dw, gs_xh, niter)
479  call profiler_end_region
480 
481  if (tstep .gt. 5 .and. params%proj_vel_dim .gt. 0) then
482  call this%proj_u%project_back(du%x, ax, c_xh, &
483  this%bclst_du, gs_xh, n)
484  call this%proj_v%project_back(dv%x, ax, c_xh, &
485  this%bclst_dv, gs_xh, n)
486  call this%proj_w%project_back(dw%x, ax, c_xh, &
487  this%bclst_dw, gs_xh, n)
488  end if
489 
490  if (neko_bcknd_device .eq. 1) then
491  call device_opadd2cm(u%x_d, v%x_d, w%x_d, &
492  du%x_d, dv%x_d, dw%x_d, 1.0_rp, n, msh%gdim)
493  else
494  call opadd2cm(u%x, v%x, w%x, du%x, dv%x, dw%x, 1.0_rp, n, msh%gdim)
495  end if
496 
497  if (params%vol_flow_dir .ne. 0) then
498  call this%vol_flow%adjust( u, v, w, p, u_res, v_res, w_res, p_res, &
499  ta1, ta2, ta3, c_xh, gs_xh, ext_bdf, params%rho, params%Re, &
500  params%dt, this%bclst_dp, this%bclst_du, this%bclst_dv, &
501  this%bclst_dw, this%bclst_vel_res, ax, this%ksp_prs, &
502  this%ksp_vel, this%pc_prs, this%pc_vel, niter)
503  end if
504 
505  call fluid_step_info(tstep, t, params%dt, ksp_results)
506 
507  end associate
508  end subroutine fluid_pnpn_step
509

511 end module fluid_pnpn
