129  subroutine scalar_scheme_add_bcs(this, zones, bc_labels)
130  class(scalar_scheme_t), intent(inout) :: this
131  type(zone_t), intent(inout) :: zones(NEKO_MSH_MAX_ZLBLS)
132  character(len=20), intent(in) :: bc_labels(NEKO_MSH_MAX_ZLBLS)
133  character(len=20) :: bc_label
134  integer :: i, j, bc_idx
135  real(kind=rp) :: dir_value
136  logical :: bc_exists
137 
138  do i = 1, neko_msh_max_zlbls
139  bc_label = trim(bc_labels(i))
140  if (bc_label(1:1) .eq. 'd') then
141  bc_exists = .false.
142  bc_idx = 0
143  do j = 1, i-1
144  if (bc_label .eq. bc_labels(j)) then
145  bc_exists = .true.
146  bc_idx = j
147  end if
148  end do
149 
150  if (bc_exists) then
151  call this%dir_bcs(j)%mark_zone(zones(i))
152  else
153  this%n_dir_bcs = this%n_dir_bcs + 1
154  call this%dir_bcs(this%n_dir_bcs)%init(this%dm_Xh)
155  call this%dir_bcs(this%n_dir_bcs)%mark_zone(zones(i))
156  read(bc_label(3:), *) dir_value
157  call this%dir_bcs(this%n_dir_bcs)%set_g(dir_value)
158  end if
159  end if
160  end do
161 
162  do i = 1, this%n_dir_bcs
163  call this%dir_bcs(i)%finalize()
164  call bc_list_add(this%bclst, this%dir_bcs(i))
165  end do
166 
167  end subroutine scalar_scheme_add_bcs
